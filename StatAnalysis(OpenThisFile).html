Data cleanup and transformation

Code 
dataset = dataset |> 
  pivot_longer(cols = control:group3,
               names_to = "treatment",
               values_to = "abs")

dataset = dataset |> 
  mutate(across(cell:treatment, as.factor)) |> 
  mutate(time = fct_relevel(time, c("24h", "48h"))) |> 
  mutate(week = fct_relevel(week, c("n1", "n2", "n3"))) |> 
  mutate(treatment = fct_relevel(treatment, c("control", 
                                              "group1", 
                                              "group2", 
                                              "group3")))
dataset = dataset |> 
  pivot_wider(names_from = "replica",
              values_from = "abs") |>
  rowwise() |>
  mutate(rep1 = median(rep1)) |>
  mutate(rep2 = median(rep2)) |>
  mutate(rep3 = median(rep3)) |>
  pivot_longer(cols = rep1:rep3,
               names_to = "replica",
               values_to = "abs") |>
  mutate(categ_final = paste0(cell, time, treatment, replica)) |>
  pivot_wider(names_from = "week",
              values_from = "abs") |>
  rowwise() |>
  mutate(abs = median(n1:n3)) |>
  select(cell:replica, abs)

Univariate Analyses

Code 
#treatment

pwc1 = dataset |>
  group_by(cell, time) |> 
  pairwise_t_test(
    abs ~ treatment, paired = FALSE,
    p.adjust.method = "bonferroni"
  )

tagList(datatable(pwc1,
                  autoHideNavigation = TRUE,
                  options = list(iDisplayLength = 25)))
Code 
#add p-values to the boxplot

pwc1 = pwc1 %>% 
  add_xy_position(x = "treatment", dodge = 0.4)

p3 = ggboxplot(dataset, 
               x = "treatment", 
               y = "abs", 
               fill = "cell",
               facet = "time") +
  stat_pvalue_manual(pwc1, 
                     label = "p.adj.signif",
                     tip.length = 0, 
                     hide.ns = TRUE) +
  scale_y_continuous(expand = expansion(mult = c(0.01, 0.1)))

p3
